#include <stdio.h>
#include <malloc.h>
#include <stdlib.h>
#include <string.h>

int compare(const void *a,const void *b) {
	return strcmp(*(char **)a,*(char **)b);
}

int main() {
	char * str[10];
	for(int i = 0;i<10;i++) {
		char * tmp = (char *)malloc(sizeof(char) * 20);
		scanf("%s",tmp);
		str[i] = tmp;
	}
	qsort(str,sizeof(str) / sizeof(str[0]),sizeof(str[0]),compare);
	for(int i = 0;i<10;i++) {
		printf("%s ",str[i]);
	}
}

#include <stdio.h>
#include <time.h>

#define day(X) X/12?"오후":"오전"

char * wday[7] = {"월","화","수","목","금","토","일"};


int main() {
	struct tm *t;

	time_t now;

	now = time(NULL);

	t = localtime(&now);

	printf("오늘은 %d년 %d월 %d일 %s요일입니다.\n",t->tm_year + 1900, t->tm_mon+1,t->tm_mday,wday[t->tm_wday]);
	printf("현재 시간은 %s %d시 %d분 %d초입니다.",day(t->tm_hour),t->tm_hour,t->tm_min,t->tm_sec);
}
