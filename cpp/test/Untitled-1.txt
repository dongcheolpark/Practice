#include <stdio.h>

typedef struct info{
	char name[20]; // 이름
	double score; // 평점 평균
	int credit; // 취득 학점
}info;

void input_data(info * students,int n) {
	for(int i = 0;i<n;i++) {
		scanf("%s %lf %d",students[i].name,&students[i].score,&students[i].credit);
	}
}

void scholarship(info * students,int n) {
	for(int i = 0;i<n;i++) {
		if(students[i].score >= 4.3 && students[i].credit >= 20) {
			printf("name : %s\nscore: %.1f\ncredit : %d\n\n",students[i].name,students[i].score,students[i].credit);
		}
	}
}

int main(void)
{
	info students[5]; // 5명의 학생
	input_data(students, 5); // 이름, 평점 평균, 취득 학점을 입력받는 함수
	scholarship(students, 5); // 장학생의 이름, 평점 평균, 취득 학점을 출력하는 함수
	return 0;
}
